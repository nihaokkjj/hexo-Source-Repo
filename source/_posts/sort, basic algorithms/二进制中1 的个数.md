~~~
#include<iostream>
using namespace std;

int lowbit(int x){
    return x&(-x);  // 返回最右边的1及其后面的0
}

int main(){
    int n;
    cin>>n;
    while(n--){
        int x;
        cin>>x;

        int res=0;
        while(x) x-=lowbit(x),res++;  // 每次减去最右边的1及其后面的0

        cout<<res<<' ';
    }
    return 0;
}
